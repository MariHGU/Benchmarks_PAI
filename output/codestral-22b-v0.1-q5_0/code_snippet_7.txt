SELECT Payments.payment_method, COUNT(*) as usage_count
FROM OrderItems
JOIN Payments ON OrderItems.order_id = Payments.order_id
WHERE OrderItems.product_id IN (
 SELECT product_id
 FROM OrderItems
 GROUP BY product_id
 ORDER BY SUM(quantity) DESC
 LIMIT 5
)
GROUP BY Payments.payment_method
ORDER BY usage_count DESC
LIMIT 1;
